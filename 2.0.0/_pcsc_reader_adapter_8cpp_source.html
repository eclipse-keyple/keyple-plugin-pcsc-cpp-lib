36void PcscReaderAdapter::waitForCardInsertion()
37{
38 mLogger->trace("%: start waiting for the insertion of a card in a loop with a latency of %" \
39 " ms\n",
40 getName(),
41 INSERTION_LATENCY);
42
43
44 /* Activate loop */
45 mLoopWaitCard = true;
46
47 try {
48 while (mLoopWaitCard) {
49 if (getTerminal()->waitForCardPresent(INSERTION_LATENCY)) {
50 /* Card inserted */
51 mLogger->trace("%: card inserted\n", getName());
52 return;
53 }
54
55 // if (Thread.interrupted()) {
56 // break;
57 // }
58 }
59 } catch (const CardException& e) {
60 /* Here, it is a communication failure with the reader */
61 throw ReaderIOException(getName() +
62 ": an error occurred while waiting for a card insertion.",
63 std::make_shared<CardException>(e));
64 }
65
66 throw TaskCanceledException(getName() +
67 ": the wait for a card insertion task has been cancelled.");
68}
