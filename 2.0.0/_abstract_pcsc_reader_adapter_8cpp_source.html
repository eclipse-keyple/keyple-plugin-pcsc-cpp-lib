97bool AbstractPcscReaderAdapter::isCurrentProtocol(const std::string& readerProtocol) const
98{
99 bool isCurrentProtocol = false;
100
101 /* C++ specific: getProtocolRule throws when protocol not found instead of returning null */
102 try {
103 const std::string protocolRule = mPluginAdapter->getProtocolRule(readerProtocol);
104 if (!protocolRule.empty()) {
105 const std::string atr = ByteArrayUtil::toHex(mTerminal->getATR());
106 isCurrentProtocol = Pattern::compile(protocolRule)->matcher(atr)->matches();
107 }
108
109 return isCurrentProtocol;
110
111 } catch (const IllegalArgumentException& e) {
112 (void)e;
113 }
114
115 return false;
116}

118void AbstractPcscReaderAdapter::openPhysicalChannel()
119{
120 /*
121 * Init of the card physical channel: if not yet established, opening of a new physical channel
122 */
123 try {
124 if (!mIsPhysicalChannelOpen) {
125 mLogger->debug("%: opening of a card physical channel for protocol '%'\n",
126 getName(),
127 mProtocol);
128 mTerminal->openAndConnect(mProtocol);
129 if (mIsModeExclusive) {
130 mTerminal->beginExclusive();
131 mLogger->debug("%: opening of a card physical channel in exclusive mode\n",
132 getName());
133 } else {
134 mLogger->debug("%: opening of a card physical channel in shared mode\n", getName());
135 }
136 }
137
138 mIsPhysicalChannelOpen = true;
139
140 } catch (const CardException& e) {
141 throw ReaderIOException(getName() + ": Error while opening Physical Channel",
142 std::make_shared<CardException>(e));
143 }
144}

183const std::vector<uint8_t> AbstractPcscReaderAdapter::transmitApdu(
184 const std::vector<uint8_t>& apduCommandData)
185{
186 std::vector<uint8_t> apduResponseData;
187
188 if (mIsPhysicalChannelOpen) {
189 try {
190 apduResponseData = mTerminal->transmitApdu(apduCommandData);
191 } catch (const CardException& e) {
192 if (e.getMessage().find("REMOVED") != std::string::npos) {
193 throw CardIOException(getName() + ":" + e.getMessage(),
194 std::make_shared<CardException>(e));
195 } else {
196 throw ReaderIOException(getName() + ":" + e.getMessage(),
197 std::make_shared<CardException>(e));
198 }
199 } catch (const IllegalStateException& e) {
200 /* Card could have been removed prematurely */
201 throw CardIOException(getName() + ":" + e.getMessage(),
202 std::make_shared<IllegalStateException>(e));
203 }
204 } else {
205 /* Could occur if the card was removed */
206 throw CardIOException(getName() + ": null channel.");
207 }
208
209 return apduResponseData;
210}

240PcscReader& AbstractPcscReaderAdapter::setSharingMode(const SharingMode sharingMode)
241{
242 mLogger->trace("%: set sharing mode to %\n", getName(), sharingMode);
243
244 if (sharingMode == SharingMode::SHARED) {
245 /* If a card is present, change the mode immediately */
246 if (mIsPhysicalChannelOpen) {
247 try {
248 mTerminal->endExclusive();
249 } catch (const CardException& e) {
250 throw IllegalStateException("Couldn't disable exclusive mode",
251 std::make_shared<CardException>(e));
252 }
253 }
254
255 mIsModeExclusive = false;
256 } else if (sharingMode == SharingMode::EXCLUSIVE) {
257 mIsModeExclusive = true;
258 }
259
260 return *this;
261}

292void AbstractPcscReaderAdapter::waitForCardRemoval()
293{
294 mLogger->trace("%: start waiting for the removal of the card in a loop with a latency of % " \
295 "ms\n",
296 getName(),
297 REMOVAL_LATENCY);
298
299
300 /* Activate loop */
301 mLoopWaitCardRemoval = true;
302
303 try {
304 while (mLoopWaitCardRemoval) {
305 if (getTerminal()->waitForCardAbsent(REMOVAL_LATENCY)) {
306 /* Card removed */
307 mLogger->trace("%: card removed\n", getName());
308 return;
309 }
310
311 // if (Thread.interrupted()) {
312 // break;
313 // }
314 }
315 } catch (const CardException& e) {
316 /* Here, it is a communication failure with the reader */
317 throw ReaderIOException(getName() +
318 ": an error occurred while waiting for the card removal.",
319 std::make_shared<CardException>(e));
320 }
321
322 throw TaskCanceledException(getName() +
323 ": the wait for the card removal task has been cancelled.");
324}
