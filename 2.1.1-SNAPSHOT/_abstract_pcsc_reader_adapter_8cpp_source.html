104bool AbstractPcscReaderAdapter::isCurrentProtocol(const std::string& readerProtocol) const
105{
106 bool isCurrentProtocol = false;
107
108 /* C++ specific: getProtocolRule throws when protocol not found instead of returning null */
109 try {
110 const std::string protocolRule = mPluginAdapter->getProtocolRule(readerProtocol);
111 if (!protocolRule.empty()) {
112 const std::string atr = HexUtil::toHex(mTerminal->getATR());
113 isCurrentProtocol = Pattern::compile(protocolRule)->matcher(atr)->matches();
114 }
115
116 return isCurrentProtocol;
117
118 } catch (const IllegalArgumentException& e) {
119 (void)e;
120 }
121
122 return false;
123}

125void AbstractPcscReaderAdapter::openPhysicalChannel()
126{
127 /*
128 * Init of the card physical channel: if not yet established, opening of a new physical channel
129 */
130 try {
131 if (!mIsPhysicalChannelOpen) {
132 mLogger->debug("%: opening of a card physical channel for protocol '%'\n",
133 getName(),
134 mProtocol);
135 mTerminal->openAndConnect(mProtocol);
136 if (mIsModeExclusive) {
137 mTerminal->beginExclusive();
138 mLogger->debug("%: opening of a card physical channel in exclusive mode\n",
139 getName());
140 } else {
141 mLogger->debug("%: opening of a card physical channel in shared mode\n", getName());
142 }
143 }
144
145 mIsPhysicalChannelOpen = true;
146
147 } catch (const CardException& e) {
148 throw ReaderIOException(getName() + ": Error while opening Physical Channel",
149 std::make_shared<CardException>(e));
150 }
151}

190const std::vector<uint8_t> AbstractPcscReaderAdapter::transmitApdu(
191 const std::vector<uint8_t>& apduCommandData)
192{
193 std::vector<uint8_t> apduResponseData;
194
195 if (mIsPhysicalChannelOpen) {
196 try {
197 apduResponseData = mTerminal->transmitApdu(apduCommandData);
198 } catch (const CardException& e) {
199 if (e.getMessage().find("REMOVED") != std::string::npos) {
200 throw CardIOException(getName() + ":" + e.getMessage(),
201 std::make_shared<CardException>(e));
202 } else {
203 throw ReaderIOException(getName() + ":" + e.getMessage(),
204 std::make_shared<CardException>(e));
205 }
206 } catch (const IllegalStateException& e) {
207 /* Card could have been removed prematurely */
208 throw CardIOException(getName() + ":" + e.getMessage(),
209 std::make_shared<IllegalStateException>(e));
210 }
211 } else {
212 /* Could occur if the card was removed */
213 throw CardIOException(getName() + ": null channel.");
214 }
215
216 return apduResponseData;
217}

247PcscReader& AbstractPcscReaderAdapter::setSharingMode(const SharingMode sharingMode)
248{
249 mLogger->trace("%: set sharing mode to %\n", getName(), sharingMode);
250
251 if (sharingMode == SharingMode::SHARED) {
252 /* If a card is present, change the mode immediately */
253 if (mIsPhysicalChannelOpen) {
254 try {
255 mTerminal->endExclusive();
256 } catch (const CardException& e) {
257 throw IllegalStateException("Couldn't disable exclusive mode",
258 std::make_shared<CardException>(e));
259 }
260 }
261
262 mIsModeExclusive = false;
263 } else if (sharingMode == SharingMode::EXCLUSIVE) {
264 mIsModeExclusive = true;
265 }
266
267 return *this;
268}

299void AbstractPcscReaderAdapter::waitForCardRemoval()
300{
301 mLogger->trace("%: start waiting for the removal of the card in a loop with a latency of % " \
302 "ms\n",
303 getName(),
304 REMOVAL_LATENCY);
305
306
307 /* Activate loop */
308 mLoopWaitCardRemoval = true;
309
310 try {
311 while (mLoopWaitCardRemoval) {
312 if (getTerminal()->waitForCardAbsent(REMOVAL_LATENCY)) {
313 /* Card removed */
314 mLogger->trace("%: card removed\n", getName());
315 return;
316 }
317
318 // if (Thread.interrupted()) {
319 // break;
320 // }
321 }
322 } catch (const CardException& e) {
323 /* Here, it is a communication failure with the reader */
324 throw ReaderIOException(getName() +
325 ": an error occurred while waiting for the card removal.",
326 std::make_shared<CardException>(e));
327 }
328
329 throw TaskCanceledException(getName() +
330 ": the wait for the card removal task has been cancelled.");
331}

358const std::vector<uint8_t> AbstractPcscReaderAdapter::transmitControlCommand(
359 const int commandId, const std::vector<uint8_t>& command)
360{
361 std::vector<uint8_t> response;
362 const int controlCode = mIsWindows ? 0x00310000 | (commandId << 2) : 0x42000000 | commandId;
363
364 try {
365 if (mTerminal != nullptr) {
366 response = mTerminal->transmitControlCommand(controlCode, command);
367
368 } else {
369 /* C++ don't do virtual cards for now... */
370 throw IllegalStateException("Reader failure.");
371
372 // std::shared_ptr<Card> virtualCard = terminal->connect("DIRECT");
373 // response = virtualCard->transmitControlCommand(controlCode, command);
374 // virtualCard->disconnect(false);
375 }
376
377 } catch (const CardException& e) {
378 throw IllegalStateException("Reader failure.",
379 std::make_shared<CardException>(e));
380 }
381
382 return response;
