123const std::string& AbstractPcscPluginAdapter::getProtocolRule(const std::string& readerProtocol)
124 const
125{
126 const auto it = mProtocolRulesMap.find(readerProtocol);
127 if (it != mProtocolRulesMap.end()) {
128 return it->second;
129 } else {
130 /*
131 * C++ specific, cannot return null here, to check if not a problem with the flow
132 * somewhere...
133 */
134 throw IllegalArgumentException("bad protocol");
135 }

138bool AbstractPcscPluginAdapter::isContactless(const std::string& readerName)
139{
140 std::unique_ptr<Pattern> p = Pattern::compile(mContactReaderIdentificationFilter);
141 if (p->matcher(readerName)->matches()) {
142 return false;
143 }
144
145 p = Pattern::compile(mContactlessReaderIdentificationFilter);
146 if (p->matcher(readerName)->matches()) {
147 return true;
148 }
149
150 throw IllegalStateException("Unable to determine the transmission mode for reader " +
151 readerName);
152}

206std::shared_ptr<ReaderSpi> AbstractPcscPluginAdapter::searchReader(const std::string& readerName)
207{
208 mLogger->trace("%: search reader: %\n", getName(), readerName);
209
210 for (const auto& terminal : getCardTerminalList()) {
211 if (readerName == terminal->getName()) {
212 mLogger->trace("%: reader: % found\n", getName(), readerName);
213 return createReader(terminal);
214 }
215 }
216
217 mLogger->trace("%: reader: % not found\n", getName(), readerName);
218
219 return nullptr;
220}
